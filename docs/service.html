var ipHandler = function(req, res){
    var ip = req.connection.remoteAddress;

    if(req.query && req.query.callback){
        res.jsonp(ip);
    }else{
        res.header('Content-Type', 'text/plain');
        res.send(ip);
    }
};
app.get('/whatsmyip', ipHandler);
